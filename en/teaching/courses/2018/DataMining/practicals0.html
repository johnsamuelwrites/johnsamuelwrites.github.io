    2. itemgetter

    from operator import itemgetter

    mots = {"contente": 12, "content": 12, "triste": 2, "joie" : 10}
    mots_tuple = [(key, value) for key,value in mots.items()]
    print(sorted(mots_tuple, key=itemgetter(1)))
