HTML5 Canvas

Histoire

  • Dévéloppé par Apple pour être utilisé dans WebKit en 2004
  • Adopté par les navigateurs utilisant Gecko en 2005
  • Adopté par Opera en 2006
  • Standardisé par le groupe de travail WHATWG
  • Il n'est pas reconnu par tous les vieux navigateurs

 document.onkeydown = function(event) {
  event = event || window.event;
  event.preventDefault();

  if (event.keyCode == '37') {
   // gauche
  }
  else if (event.keyCode == '38') {
   // haut
  }
  else if (event.keyCode == '39') {
   // droite
  }
  else if (event.keyCode == '40') {
   // bas
  }
 }
